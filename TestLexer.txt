/* Ejemplo de gramática en notación BNF */
 Expr ::= Term
         | Expr <PLUS> Term
		 ;
Term ::= Factor
         | Term <PROD> Factor
		 ;
  /* Los factores son constantes, expresiones entre paréntesis o llamadas a funciones */
  Factor ::= <NUM>
         | <LPAREN> Expr <RPAREN>
		 ;
<Elementos> ::= A | a | _ | cd | _5 | as6 | fas7a | 832 | _9aBa_
